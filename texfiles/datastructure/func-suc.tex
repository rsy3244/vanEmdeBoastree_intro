\begin{frame}\frametitle{操作$\func{successor}(e)$, $\func{predecessor}(e)$}
\begin{itemize}
	\item $\func{successor}(e)$は, 集合の要素から$e$より大きい最小の値を返す.\\
	\item $\func{predecessor}(e)$は, 集合の要素から$e$より小さい最大の値を返す.\\
\end{itemize}
