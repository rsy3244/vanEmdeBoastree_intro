\begin{frame}\frametitle{操作$\Successor(V,x), \Predecessor(V,x)$}
\begin{itemize}
	\item $\Successor(V,x)$は, 集合$V$の要素から$x$より大きい最小の値を返す.\\
	\item $\Predecessor(V,x)$は, 集合$V$の要素から$x$より小さい最大の値を返す.\\
\end{itemize}

		\end{itemize}
	\end{column}
