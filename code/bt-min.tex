\begin{algorithm}[H]
\caption{$\Min(V)$}
\begin{algorithmic}[1]
\scriptsize
\Require
	\Statex	$V :=$ 二分木を格納した配列
	\Statex $u :=$ 全体集合の大きさ
\Function{$\Min$}{$V$}
	\State $i \gets 1$
	\While{$i < u$}
		\If{V[$2\times i + 1$] = 1}
			\State $i \gets 2\times i + 1$
		\ElsIf{V[$2 \times i + 2$] = 1}
			\State $i \gets 2\times i + 2$
		\Else
			\State \Return $\NIL$
		\EndIf
	\EndWhile
	\State \Return $i - u$
\EndFunction
\end{algorithmic}
\end{algorithm}
